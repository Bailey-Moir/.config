$$ 
\begin{array}{c}
	(f \circ g)(x) = f(g(x)) \\
	\begin{array}{l l}
		(f+g)(x) = f(x) + g(x) & (fg)(x) = f(x)g(x) \\
		
			(f-g)(x) = f(x) - g(x) & \left(\dfrac{f}{g}\right)(x) = \dfrac{f(x)}{g(x)} \\
	\end{array}
\end{array}
$$
