$$\begin{align*}
	\{a_n\} \text{ bounded above} &\iff \exists M \ni (n \ge 1 \implies a_n<M) \\
	\{a_n\} \text{ bounded below} &\iff \exists N \ni (n \ge 1 \implies N<a_n) \\
	\{a_n\} \text{ bounded} &\iff \exists N,M \ni (n \ge 1 \implies N<a_n<M)
\end{align*}$$
